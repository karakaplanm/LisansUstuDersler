import numpy as np
   
x = np.linspace(0,10,50)
f1 = np.sin(x)**2
f2 = np.cos(x**2)
import pylab

pylab.rc('text', usetex=True)
pylab.rc('font', family='serif')

pylab.plot(x,f1,label=r'$f(x)=sin^{2}(\sigma)$',linestyle='dashed')
pylab.plot(x,f2,label=r'$g(x)=cos(\sigma^{2})$',linestyle='dashdot')
pylab.plot(x,f1/f2,label=r'$\frac{f(x)}{g(x)}$',linestyle='solid')

pylab.xlabel(r'$\sigma$',fontsize=20)
pylab.ylabel('y', fontsize=20)
pylab.tick_params(labelsize=20)
pylab.legend(fontsize=20)
pylab.savefig("fonk.pdf")
